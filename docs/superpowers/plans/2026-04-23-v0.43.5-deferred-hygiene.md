```python
import os
import tempfile
from contextlib import contextmanager


@contextmanager
def _atomic_write_csv(path):
    """Yield a writable file handle whose contents are atomically renamed
    to `path` on successful close. A killed process never leaves a
    half-written CSV visible to readers.

    Uses tempfile.NamedTemporaryFile in the same directory so os.replace
    is a same-volume rename (atomic on POSIX and Windows).
    """
    path = Path(path)
    path.parent.mkdir(parents=True, exist_ok=True)
    tmp = tempfile.NamedTemporaryFile(
        mode="w",
        dir=str(path.parent),
        prefix=f".{path.name}.",
        suffix=".tmp",
        delete=False,
        newline="",
        encoding="utf-8",
    )
    try:
        yield tmp
        tmp.flush()
        os.fsync(tmp.fileno())
        tmp.close()
        os.replace(tmp.name, str(path))
    except Exception:
        tmp.close()
        try:
            os.unlink(tmp.name)
        except OSError:
            pass
        raise
```

def apply_superimposition(redd_state, new_redd_idx: int, redd_area: float, cell_area: float) -> None:
    """Destroy a fraction of eggs in existing redds that overlap the new redd.

    NetLogo InSALMO behavior: fraction destroyed = min(1.0, redd_area / cell_area).
    Previously a hardcoded 50% regardless of overlap.
    """
    if cell_area <= 0:
        return
    fraction = min(1.0, redd_area / cell_area)
    new_cell = redd_state.cell_idx[new_redd_idx]

        if i == new_redd_idx or not redd_state.alive[i]:
            continue
        if redd_state.cell_idx[i] == new_cell:
            redd_state.num_eggs[i] = int(redd_state.num_eggs[i] * (1.0 - fraction))
